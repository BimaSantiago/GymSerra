DROP DATABASE IF EXISTS gym_serra_1;
CREATE DATABASE gym_serra_1;
USE gym_serra_1;
SET GLOBAL event_scheduler = ON;

CREATE TABLE unidad_medida (
  idunidad INT AUTO_INCREMENT PRIMARY KEY,
  clave VARCHAR(10) NOT NULL UNIQUE,
  descripcion VARCHAR(100) NOT NULL,
  tipo VARCHAR(30)
);
CREATE TABLE categorias (
  idcategoria INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(150) NOT NULL,
  id_padre INT NULL,
  orden INT UNSIGNED NOT NULL DEFAULT 0,
  FOREIGN KEY (id_padre) REFERENCES categorias(idcategoria) ON DELETE SET NULL ON UPDATE CASCADE
);
CREATE TABLE articulos (
  idarticulo INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(100) NOT NULL,
  codigo_barras VARCHAR(100),
  descripcion TEXT,
  descripcion2 ENUM('Mobiliario','Venta') DEFAULT 'Mobiliario',
  stock INT DEFAULT 0,
  estado ENUM('Activo','Inactivo') DEFAULT 'Activo',
  ganancia DECIMAL(10,2),
  iva_aplicable ENUM('Si', 'No') DEFAULT 'Si',
  idunidad INT NOT NULL,
  idcategoria INT NOT NULL,
  img VARCHAR(255) DEFAULT NULL,
  PRIMARY KEY(idarticulo),
  FOREIGN KEY (idunidad) REFERENCES unidad_medida(idunidad),
  FOREIGN KEY (idcategoria) REFERENCES categorias(idcategoria)
);
CREATE TABLE lista_precio (
  idprecio INT NOT NULL AUTO_INCREMENT,
  idarticulo INT NOT NULL,
  precio DECIMAL(10,2) NOT NULL,
  fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (idprecio),
  FOREIGN KEY (idarticulo) REFERENCES articulos(idarticulo)
);

CREATE TABLE lista_costo (
  idcosto INT NOT NULL AUTO_INCREMENT,
  idarticulo INT NOT NULL,
  precio DECIMAL(10,2) NOT NULL,
  fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (idcosto),
  FOREIGN KEY (idarticulo) REFERENCES articulos(idarticulo)
);

CREATE TABLE proveedores (
  idproveedor INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(50) NOT NULL,
  RFC VARCHAR(150) NOT NULL,
  codigo_postal VARCHAR(150),
  calle VARCHAR(250) NOT NULL,
  localidad VARCHAR(150) NOT NULL,
  municipio VARCHAR(150) NOT NULL,
  telefono VARCHAR(20) NOT NULL,
  correo VARCHAR(150) NOT NULL,
  estado VARCHAR(20) NOT NULL,
  nombre_vendedor VARCHAR(150) NOT NULL,
  telefono_vendedor VARCHAR(15) NOT NULL,
  correo_vendedor VARCHAR(100) NOT NULL,
  PRIMARY KEY (idproveedor)
);

CREATE TABLE articulo_proveedor (
  idarticulo INT NOT NULL,
  idproveedor INT NOT NULL,
  PRIMARY KEY (idarticulo, idproveedor),
  FOREIGN KEY (idarticulo) REFERENCES articulos(idarticulo),
  FOREIGN KEY (idproveedor) REFERENCES proveedores(idproveedor)
);

CREATE TABLE cliente (
  idcliente INT NOT NULL AUTO_INCREMENT,
  curp VARCHAR(50) NOT NULL,
  nombre_completo VARCHAR(150) NOT NULL,
  f_nacimiento DATE NOT NULL,
  estado ENUM('Activo', 'Inactivo') DEFAULT 'Activo',
  fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
  f_ultima_compra DATETIME DEFAULT NULL,
  PRIMARY KEY (idcliente)
);

CREATE TABLE users (
  iduser INT NOT NULL AUTO_INCREMENT,
  username VARCHAR(100) NOT NULL,
  passw VARCHAR(100) NOT NULL,
  avatar VARCHAR(100),
  nombre VARCHAR(50) NOT NULL,
  appaterno VARCHAR(50) NOT NULL,
  apmaterno VARCHAR(50) NOT NULL,
  correo VARCHAR(100) NOT NULL,
  genero VARCHAR(15) NOT NULL,
  rol ENUM('Administrador','Estandar') DEFAULT 'Estandar',
  estatus ENUM('activo','inactivo','bloqueado') DEFAULT 'activo',
  intentos_fallidos INT DEFAULT 0,
  ultimo_intento_login DATETIME NULL,
  bloqueado_hasta DATETIME NULL,
  fecha_ultimo_login DATETIME NULL,
  fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (iduser)
);

CREATE TABLE corte_caja (
  idcorte INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  folio VARCHAR(30) NOT NULL,
  fecha_inicio DATETIME DEFAULT CURRENT_TIMESTAMP,
  fecha_corte DATETIME,
  total_vendido DECIMAL(10,2),
  tipo ENUM('X','Y') DEFAULT 'X',
  corte_principal INT,
  CONSTRAINT FK_cortes
        FOREIGN KEY (corte_principal)
        REFERENCES corte_caja (idcorte)
);

CREATE TABLE movimiento(
  idmovimiento INT NOT NULL AUTO_INCREMENT,
  comentario VARCHAR(100),
  fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
  total DECIMAL(10,2),
  iduser INT,
  idcliente INT,
  idcorte INT,
  idproveedor INT,  
  tipo ENUM('Entrada','Salida','Compra','Venta') DEFAULT NULL,
  PRIMARY KEY (idmovimiento),
  FOREIGN KEY (iduser) REFERENCES users(iduser),
  FOREIGN KEY (idcliente) REFERENCES cliente(idcliente),
  FOREIGN KEY (idcorte) REFERENCES corte_caja(idcorte),
  FOREIGN KEY (idproveedor) REFERENCES proveedores(idproveedor)

);
CREATE TABLE detalle_movimiento (
  iddetalle_movimiento INT NOT NULL AUTO_INCREMENT,
  idmovimiento INT NOT NULL,
  idarticulo INT NOT NULL,
  conteo INT,
  diferencia INT,
  subtotal DECIMAL(10,2),
  cantidad INT,
  idcosto INT,
  idprecio INT,
  PRIMARY KEY (iddetalle_movimiento),
  FOREIGN KEY (idmovimiento) REFERENCES movimiento(idmovimiento),
  FOREIGN KEY (idarticulo) REFERENCES articulos(idarticulo),
  FOREIGN KEY (idprecio) REFERENCES lista_precio(idprecio), 
  FOREIGN KEY (idcosto) REFERENCES lista_costo(idcosto)

);
CREATE TABLE motivos_cancelacion (
  idmotivo INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(50) NOT NULL,
  tipo ENUM('Devolucion','Cancelacion') DEFAULT 'Cancelacion'
);

CREATE TABLE devolucion (
  iddevolucion INT NOT NULL AUTO_INCREMENT,
  idmovimiento INT NOT NULL,
  monto_devuelto DECIMAL(10,2) NOT NULL DEFAULT 0,
  fecha_devolucion DATETIME DEFAULT CURRENT_TIMESTAMP,
  idcorte INT NOT NULL,
  PRIMARY KEY (iddevolucion),
  FOREIGN KEY (idmovimiento) REFERENCES movimiento(idmovimiento),
  FOREIGN KEY (idcorte) REFERENCES corte_caja(idcorte)
);

CREATE TABLE devolucion_detalle (
  iddevolucion_detalle INT NOT NULL AUTO_INCREMENT,
  iddevolucion INT NOT NULL,
  idarticulo INT NOT NULL,
  idprecio INT NOT NULL,
  idmotivo INT NOT NULL,
  cantidad_devuelta INT NOT NULL,
  subtotal DECIMAL(10,2),
  PRIMARY KEY (iddevolucion_detalle),
  FOREIGN KEY (iddevolucion) REFERENCES devolucion(iddevolucion),
  FOREIGN KEY (idarticulo) REFERENCES articulos(idarticulo),
  FOREIGN KEY (idprecio) REFERENCES lista_precio(idprecio),
  FOREIGN KEY (idmotivo) REFERENCES motivos_cancelacion(idmotivo)
);

CREATE TABLE cancelacion (
  idcancelacion INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  idmovimiento INT DEFAULT NULL,
  idmotivo INT NOT NULL,
  descripcion VARCHAR(255) NOT NULL,
  FOREIGN KEY (idmovimiento) REFERENCES movimiento(idmovimiento),
  FOREIGN KEY (idmotivo) REFERENCES motivos_cancelacion(idmotivo)
);

CREATE TABLE deporte (
  iddeporte INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(50) NOT NULL,
  descripcion VARCHAR(150) NOT NULL,
  color VARCHAR(20) NOT NULL,
  PRIMARY KEY (iddeporte)
);

CREATE TABLE nivel (
  idnivel INT NOT NULL AUTO_INCREMENT,
  iddeporte INT NOT NULL,
  nombre_nivel VARCHAR(50) NOT NULL,
  PRIMARY KEY (idnivel),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte)
);

CREATE TABLE horarios (
  idhorario INT NOT NULL AUTO_INCREMENT,
  hora_inicio INT NOT NULL,
  hora_fin INT NOT NULL,
  dia INT NOT NULL,
  idnivel INT NOT NULL,
  iddeporte INT NOT NULL,
  PRIMARY KEY (idhorario),
  FOREIGN KEY (idnivel) REFERENCES nivel(idnivel),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte)
);

CREATE TABLE plan_pago (
  idplan INT NOT NULL AUTO_INCREMENT,
  iddeporte INT NOT NULL,
  dias_por_semana INT NOT NULL,
  costo DECIMAL(10,2) NOT NULL,
  costo_promocion DECIMAL(10,2) NOT NULL,
  costo_penalizacion DECIMAL(10,2) NOT NULL,
  PRIMARY KEY (idplan),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte)
);

CREATE TABLE tutores (
  idtutor INT NOT NULL AUTO_INCREMENT,
  nombre_completo VARCHAR(150) NOT NULL,
  curp VARCHAR(50) NOT NULL,
  telefono VARCHAR(20) NOT NULL,
  correo VARCHAR(100) NOT NULL,
  estado_documentos ENUM('Completo', 'Incompleto') DEFAULT 'Incompleto',
  PRIMARY KEY (idtutor)
);

CREATE TABLE alumnos (
  idalumno INT NOT NULL AUTO_INCREMENT,
  idtutor INT,
  curp VARCHAR(50) NOT NULL UNIQUE,
  nombre_completo VARCHAR(150) NOT NULL,
  f_nacimiento DATE NOT NULL,
  estado ENUM('Activo', 'Inactivo') DEFAULT 'Activo',
  estado_documentos ENUM('Completo', 'Incompleto') DEFAULT 'Completo',
  fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (idalumno),
  FOREIGN KEY (idtutor) REFERENCES tutores(idtutor)
);

CREATE TABLE mensualidad (
  idmensualidad INT NOT NULL AUTO_INCREMENT,
  idalumno INT NOT NULL,
  fecha_pago DATE DEFAULT '2025-01-01',
  idplan INT NOT NULL,
  total_pagado DECIMAL(10,2) NOT NULL,
  estado ENUM('Pagado','Pendiente','Vencido') DEFAULT 'Pendiente',
  PRIMARY KEY (idmensualidad),
  FOREIGN KEY (idplan) REFERENCES plan_pago(idplan),
  FOREIGN KEY (idalumno) REFERENCES alumnos(idalumno)
);

DELIMITER $$

CREATE EVENT IF NOT EXISTS actualizar_mensualidades_vencidas
ON SCHEDULE EVERY 1 MONTH
STARTS TIMESTAMP(CURRENT_DATE + INTERVAL (1 - DAY(CURRENT_DATE)) DAY)
DO
BEGIN
  UPDATE mensualidad
  SET estado = 'Vencido'
  WHERE estado != 'Pagado';
END$$

DELIMITER ;

CREATE TABLE eventos (
  idevento INT NOT NULL AUTO_INCREMENT,
  fecha_inicio DATE,
  fecha_fin DATE,
  descripcion VARCHAR(100) NOT NULL,
  iddeporte INT NOT NULL,
  PRIMARY KEY (idevento),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte)
);

CREATE TABLE noticias (
  idnoticias INT NOT NULL AUTO_INCREMENT,
  titulo VARCHAR(50) NOT NULL,
  fecha_publicacion DATETIME DEFAULT CURRENT_TIMESTAMP,
  iddeporte INT NOT NULL,
  idevento INT NOT NULL,
  descripcion VARCHAR(150) NOT NULL,
  imagen VARCHAR(150) NOT NULL,
  PRIMARY KEY (idnoticias),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte),
  FOREIGN KEY (idevento) REFERENCES eventos(idevento)
);

CREATE TABLE clase_prueba (
  idclase_prueba INT NOT NULL AUTO_INCREMENT,
  idalumno INT NOT NULL UNIQUE,
  idhorario INT NOT NULL,
  fecha_clase DATE NOT NULL,
  estado ENUM('Programada','Tomada') DEFAULT 'Programada',
  PRIMARY KEY (idclase_prueba),
  FOREIGN KEY (idalumno) REFERENCES alumnos(idalumno),
  FOREIGN KEY (idhorario) REFERENCES horarios(idhorario)
);

CREATE TABLE contactos (
  idcontacto INT NOT NULL AUTO_INCREMENT,
  iddeporte INT NOT NULL,
  nombre_tutor VARCHAR(150) NOT NULL,
  nombre_alumno VARCHAR(150) NOT NULL,
  edad INT NOT NULL,
  telefono VARCHAR(15) NOT NULL,
  mensaje VARCHAR(255) NOT NULL,
  PRIMARY KEY (idcontacto),
  FOREIGN KEY (iddeporte) REFERENCES deporte(iddeporte)
);

CREATE TABLE instructores(
  idinstructor INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  iddeporte INT NOT NULL,
  nombre VARCHAR(50) NOT NULL,
  appaterno VARCHAR(50) NOT NULL,
  apmaterno VARCHAR(50) NOT NULL,
  telefono VARCHAR(20) NOT NULL,
  correo VARCHAR(255) NOT NULL,
  estado ENUM('Activo','Inactivo') DEFAULT 'Activo',
  fecha_creacion DATE,
  avatar VARCHAR(250)
);
